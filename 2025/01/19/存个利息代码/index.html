存个利息代码

发表于 2025-01-19

这只是一个存档，用来存储我的支付宝的小荷包的计划收益

目前想的是一天1元开始0.03元累加的稳定存储，先这样开始吧，运行代码看看一年后200元/分的收益+每个月200的固定存储+每日存储看看一年后能存多少钱。
啊，如果有什么奖金啊我也会存进去的，看看能不能攒出一个switch2或者其他电子产品什么的…
或者明年买GTA6的钱？或者下一张GIDLE专辑的钱？
先一年开始吧，留点钱比什么都有用。

#include<bits/stdc++.h>
using namespace std;

int main() {
    system("chcp 65001");
    double initial_input = 1;  // 初始存入金额
    double total_sum = 0;      // 总存款
    int month = 1;             // 当前月份
    double monthly_input = 0;  // 本月存款
    double remain = 0;         // 剩余余额
    double per_rate = 0.03;    // 每月递增比例

    double monthly_expense = 2000 - 150 - 200;  // 每月固定支出（生活费、订阅费等）
    double daily_food_expense = (5 + 18 + 20) * 30;  // 每月餐费支出

    printf("初始存入金额为 %.2f, 每月递增比率为 %.2f\n", initial_input, per_rate);
    cout << endl;

    bool remain_decreases = false;  // 用于标记remain是否开始减少
    bool negative_remain = false;   // 用于标记是否出现负余额

    // 模拟365天
    for(int i = 0; i < 365; i++) {
        total_sum += initial_input;  // 总存款累计
        monthly_input += initial_input;  // 本月存款累计
        initial_input += per_rate;  // 增加递增金额
        // 计算收益：每200元存款产生0.01元收益
        double current_interest = floor(total_sum / 200) * 0.01;  // 满200元的部分计算收益
        total_sum+=current_interest; // 收益加到存款里


        // 每个月结束后，计算剩余余额
        if (i % 30 == 0 && i != 0) {
            total_sum+=200;//每月存入两百
            double current_remain = monthly_expense - monthly_input - daily_food_expense;  // 当前月剩余金额
            remain += current_remain;  // 累计剩余余额

            // 检查剩余余额是否减少
            if (!remain_decreases && remain > 0 && current_remain < 0) {
                remain_decreases = true;
                printf("从第 %d 个月开始，剩余余额开始减少。\n", month);
            }

            // 检查余额是否变为负数
            if (!negative_remain && remain < 0) {
                negative_remain = true;
                printf("第 %d 个月，剩余余额变为负数。\n", month);
            }

            cout << "第 " << month << " 个月的存入金额为: " << initial_input << " 本月所有存入总额为: " << monthly_input << endl;
            printf("本月餐费为 %.2f\n", monthly_expense - monthly_input);
            printf("剩余余额为 %.2f\n", remain);
            
            printf("目前存入总金额为 %.2f\n",total_sum+0.04);
            cout << endl;

            month++;
            monthly_input = 0;  // 重置本月存款
        }
    }

    //printf("总存款金额为: %.2f\n", total_sum);
    printf("本年度未包装的支出为: %.2f\n", 2000 * 12 - total_sum);
    printf("每日存入金额最终为: %.2f\n", initial_input);
    printf("最终剩余余额为: %.2f\n", remain);

    return 0;
}
